    <script>
        // --- Navigation Scroll Spy ---
        const sections = document.querySelectorAll("section");
        const navLinks = document.querySelectorAll(".nav-link");

        window.addEventListener("scroll", () => {
            let current = "";
            sections.forEach((section) => {
                const sectionTop = section.offsetTop;
                if (scrollY >= sectionTop - 100) {
                    current = section.getAttribute("id");
                }
            });

            navLinks.forEach((link) => {
                link.classList.remove("active", "border-b-2", "border-gold-500", "text-gold-500");
                link.classList.add("text-warm-700");
                if (link.getAttribute("href").includes(current)) {
                    link.classList.add("active");
                    link.style.borderBottom = "2px solid #d9a05b";
                    link.style.color = "#d9a05b";
                } else {
                    link.style.borderBottom = "none";
                }
            });
        });

        // --- Mechanics Stepper Logic ---
        function showStep(stepNum) {
            // Reset buttons
            document.querySelectorAll('.step-btn').forEach(btn => {
                btn.classList.remove('border-gold-500', 'bg-white', 'shadow-md');
                btn.classList.add('border-transparent', 'bg-warm-50');
                
                // Reset circle color
                const circle = btn.querySelector('div > div');
                circle.classList.remove('bg-gold-500', 'text-white');
                circle.classList.add('bg-warm-200', 'text-warm-700');
            });

            // Reset contents
            document.querySelectorAll('.step-content').forEach(content => {
                content.classList.add('hidden');
            });

            // Activate chosen button
            const activeBtn = document.getElementById(`btn-step-${stepNum}`);
            activeBtn.classList.remove('border-transparent', 'bg-warm-50');
            activeBtn.classList.add('border-gold-500', 'bg-white', 'shadow-md');
            
            const activeCircle = activeBtn.querySelector('div > div');
            activeCircle.classList.remove('bg-warm-200', 'text-warm-700');
            activeCircle.classList.add('bg-gold-500', 'text-white');

            // Show chosen content
            const activeContent = document.getElementById(`content-step-${stepNum}`);
            activeContent.classList.remove('hidden');
            // Re-trigger animation
            activeContent.style.animation = 'none';
            activeContent.offsetHeight; /* trigger reflow */
            activeContent.style.animation = 'fadeIn 0.3s ease-in-out';
        }

        // --- Coaching Tabs Logic ---
        function openTab(tabId) {
            // Hide all contents
            document.querySelectorAll('.tab-content').forEach(tab => {
                tab.classList.remove('active');
            });
            // Reset all buttons
            document.querySelectorAll('[id^="btn-"]').forEach(btn => {
                if(btn.id.includes('tab')) {
                    btn.classList.remove('bg-gold-500', 'text-white', 'shadow-md');
                    btn.classList.add('bg-warm-200', 'text-warm-900');
                }
            });

            // Show target content
            document.getElementById(tabId).classList.add('active');
            
            // Highlight active button
            const activeBtn = document.getElementById(`btn-${tabId}`);
            activeBtn.classList.remove('bg-warm-200', 'text-warm-900');
            activeBtn.classList.add('bg-gold-500', 'text-white', 'shadow-md');
        }

        // --- Chart.js Financial Projections Logic ---
        const ctx = document.getElementById('financialChart').getContext('2d');
        
        // Base Data Variables
        const rewardPerStar = 20;
        const rates = [50, 60, 70, 80, 90, 100]; // 5-Star Percentages
        
        // Function to calculate costs based on volume
        function calculateCosts(volume) {
            return rates.map(rate => {
                const starsEarned = volume * (rate / 100);
                return starsEarned * rewardPerStar;
            });
        }

        // Initial Chart Setup
        let initialVolume = 100;
        let chartConfig = {
            type: 'bar',
            data: {
                labels: rates.map(r => r + '% 5-Star Rate'),
                datasets: [{
                    label: 'Projected Cost (₱) per Team',
                    data: calculateCosts(initialVolume),
                    backgroundColor: '#d9a05b', // gold-500
                    borderRadius: 6,
                    hoverBackgroundColor: '#c25934' // alert-500 highlight
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: {
                        display: false
                    },
                    tooltip: {
                        callbacks: {
                            label: function(context) {
                                return '₱ ' + context.parsed.y.toLocaleString();
                            }
                        },
                        backgroundColor: '#3d342b',
                        titleFont: { family: 'Inter', size: 14 },
                        bodyFont: { family: 'Inter', size: 14, weight: 'bold' },
                        padding: 12
                    }
                },
                scales: {
                    y: {
                        beginAtZero: true,
                        grid: { color: '#e6decb' },
                        ticks: {
                            font: { family: 'Inter', color: '#5c4e40' },
                            callback: function(value) {
                                return '₱' + value;
                            }
                        }
                    },
                    x: {
                        grid: { display: false },
                        ticks: { font: { family: 'Inter', color: '#5c4e40' } }
                    }
                }
            }
        };

        const financialChart = new Chart(ctx, chartConfig);

        // Interaction: Range Slider updates Chart
        const volumeSlider = document.getElementById('surveyVolume');
        const volumeValueDisplay = document.getElementById('volumeValue');

        volumeSlider.addEventListener('input', function(e) {
            const currentVolume = parseInt(e.target.value);
            volumeValueDisplay.textContent = currentVolume;
            
            // Update chart data
            financialChart.data.datasets[0].data = calculateCosts(currentVolume);
            financialChart.update();
        });

    </script>
